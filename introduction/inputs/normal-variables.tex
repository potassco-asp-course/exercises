\begin{Uebung}{(Normal Logic Programs with Variables)}%
Determine 
the ground instantiation
and the stable models of the following
normal logic programs with variables.
\begin{UList}
\item
\(
P =
\left\{
\begin{array}{r@{{}\leftarrow{}}l}
    \mathit{sparrow}(\mathit{jack})        \\
    \mathit{penguin}(\mathit{tweety})      \\
    \mathit{bird}(X)  & \mathit{sparrow}(X) \\
    \mathit{bird}(X)  & \mathit{penguin}(X) \\
    \mathit{flies}(X) & \mathit{bird}(X), \naf{\mathit{penguin}(X)}
\end{array}
\right\}
\)
\item 
\(
P =
\left\{
\begin{array}{r@{{}\leftarrow{}}l}
    \mathit{next}(0,1) \\
    \mathit{next}(1,0) \\
    \mathit{even}(X) & \naf{\mathit{odd}(X)} \\
    \mathit{odd}(Y)  & \mathit{next}(X,Y), \mathit{even}(X)
\end{array}
\right\}
\)
\item
\(
P =
\left\{
\begin{array}{r@{{}\leftarrow{}}l}
    \mathit{next}(0,1) \\
    \mathit{next}(1,2) \\
    \mathit{next}(2,0) \\
    \mathit{select}(X)   & \mathit{next}(X,Y), \naf{\mathit{select}(Y)}
\end{array}
\right\}
\)
\item
\(
P =
\left\{
\begin{array}{r@{{}\leftarrow{}}l}
    \mathit{friend}(\mathit{alice},\mathit{bob})  \\
    \mathit{friend}(\mathit{bob},\mathit{alice})  \\
    \mathit{friend}(\mathit{eve},\mathit{alice})  \\
    \mathit{refuse}(X) & \naf{\mathit{invite}(X)} \\
    \mathit{invite}(X) & \naf{\mathit{refuse}(X)} \\
    \mathit{invite}(Y) & \mathit{invite}(X), \mathit{friend}(X,Y)
\end{array}
\right\}
\)
\end{UList}
\end{Uebung}

\begin{Loesung}%
\textnormal{%
\begin{UList}
\item
\(
ground(P) =
\left\{
\begin{array}{r@{{}\leftarrow{}}l}
    \mathit{sparrow}(\mathit{jack})        \\
    \mathit{penguin}(\mathit{tweety})      \\
    \mathit{bird}(jack)  & \mathit{sparrow}(jack) \\
    \mathit{bird}(tweety)  & \mathit{sparrow}(tweety) \\
    \mathit{bird}(jack)  & \mathit{penguin}(jack) \\
    \mathit{bird}(tweety)  & \mathit{penguin}(tweety) \\
    \mathit{flies}(jack) & \mathit{bird}(jack), \naf{\mathit{penguin}(jack)}\\
    \mathit{flies}(tweety) & \mathit{bird}(tweety), \naf{\mathit{penguin}(tweety)}
\end{array}
\right\}
\)
\\[3mm]
Stable Models:
\begin{itemize}
\item $\{\mathit{sparrow}(\mathit{jack}),
         \mathit{penguin}(\mathit{tweety}),
         \mathit{bird}(\mathit{jack}),
         \mathit{bird}(\mathit{tweety}),
         \mathit{flies}(\mathit{jack})\}$
\end{itemize}
\item
\(
ground(P) =
\left\{
\begin{array}{r@{{}\leftarrow{}}l}
    \mathit{next}(0,1) \\
    \mathit{next}(1,0) \\
    \mathit{even}(0) & \naf{\mathit{odd}(0)} \\
    \mathit{even}(1) & \naf{\mathit{odd}(1)} \\
    \mathit{odd}(0)  & \mathit{next}(0,0), \mathit{even}(0)\\
    \mathit{odd}(1)  & \mathit{next}(0,1), \mathit{even}(0)\\
    \mathit{odd}(0)  & \mathit{next}(1,0), \mathit{even}(1)\\
    \mathit{odd}(1)  & \mathit{next}(1,1), \mathit{even}(1)
\end{array}
\right\}
\)
\\[3mm]
Stable Models:
\begin{itemize}
\item $\{\mathit{next}(0,1),\mathit{next}(1,0),\mathit{even}(0),\mathit{odd}(1)\}$
\item $\{\mathit{next}(0,1),\mathit{next}(1,0),\mathit{even}(1),\mathit{odd}(0)\}$
\end{itemize}
\item
\(
ground(P) =

    \mathit{select}(2)   & \mathit{next}(2,1), \naf{\mathit{select}(1)} \\

Stable Models: None

    \mathit{refuse}(eve) & \naf{\mathit{invite}(eve)} \\

    \mathit{invite}(bob) & \naf{\mathit{refuse}(bob)} \\

       \right\}$
\item $\left\{
       \begin{array}{@{}l@{}}
         \mathit{friend}(\mathit{alice},\mathit{bob}),
         \mathit{friend}(\mathit{bob},\mathit{alice}),
         \mathit{friend}(\mathit{eve},\mathit{alice}),
       {} \\
         \mathit{invite}(\mathit{alice}),
         \mathit{invite}(\mathit{bob}),
         \mathit{invite}(\mathit{eve})
       \end{array}
       \right\}$
\end{itemize}
\end{UList}}
\end{Loesung}
