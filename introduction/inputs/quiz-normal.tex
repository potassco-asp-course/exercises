\begin{quiz}
\quizAufgabe{Each normal logic program has some model.}
\quizAufgabe{Each normal logic program has some stable model.}
\quizAufgabe{Given a normal logic program~$P$,
             the reduct $\reduct{P}{X}$ is contained in $\reduct{P}{Y}$
             for each subset $X$ of a set $Y$ of atoms.}
\quizAufgabe{If a set $X$ of atoms is a model of a normal logic program~$P$,
             then $X$ is a model of the reduct $\reduct{P}{Y}$
             for each superset~$Y$ of~$X$.}
\quizAufgabe{Given a normal logic program~$P$,
             each stable model of~$P$ is a $\subseteq$-minimal model of~$P$.}
\end{quiz}
